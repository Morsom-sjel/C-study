**1、判断（公式）中，相等是两个等号“==”
2、同赢同输均不得分
3、符合条件，对方++**

**我的代码：**

```
#include<stdio.h>

int main()
{
	int N;
	int i,j,k,l,m;
	int a=0,b=0;
	scanf("%d",&N);
	for(i=0;i<N;i++)
	{
		scanf("%d %d %d %d",&j,&k,&l,&m);
		if(k==j+l&&m!=j+l)
		{
			a++;
		}
		if(m==j+l&&k!=j+l)
		{
			b++;
		}
	}
	
	printf("%d %d",b,a);
	return 0;
}
```
