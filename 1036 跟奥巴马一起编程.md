**1、输出字符：%c
    2、if里的判断 ‘等于’ “==”（两个等号）**
    
**我的代码：**

```
#include<stdio.h>

int main()
{
	int i,j,k,N;
	char C;
	scanf("%d",&N);
	scanf("%s",&C);
	for (i = 0; i <(N+1)/2; i++)
	{
		for(j=0;j<N;j++)
		{
			if (i==0||i==(N-1)/2||j==0||j==N-1)
			{
				printf("%c",C);
			}else
			{
				printf(" ");
			}
		}
		printf("\n");
	}
	return 0;
}
```
