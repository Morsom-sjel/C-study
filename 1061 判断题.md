#include<stdio.h>

int main()
{
	int n,i,j,m,sum;
	scanf("%d %d",&n,&m);
	int fs[100];
	int da[100];
	int xs[100];
	int jg[100];
	for(i=0;i<m;i++){
		scanf("%d ",&fs[i]);
	}
	for(i=0;i<m;i++){
		scanf("%d ",&da[i]);
	}
	for(i=0;i<n;i++){
		sum=0;
		for(j=0;j<m;j++){
			scanf("%d",&xs[j]);
			if(xs[j]==da[j]){
				sum+=fs[j];
			}
		}
		jg[i]=sum;
	}
	for(i=0;i<n-1;i++){
		printf("%d\n",jg[i]);
	}
	printf("%d",jg[n-1]);
	return 0;
}
