**我的代码：**

```
#include<stdio.h>

int main()
{
	int N,i,j;
	int a[10]={0},sum=0;
	scanf("%d",&N);
    for(i=0;i<N;i++)
	{
        scanf("%d",&a[i]);
	}
    for(i=0;i<N;i++)
    {
        for(j=0;j<N;j++)
        {
            if(j!=i)
                sum=sum+a[i]*10+a[j];
        }
    }
    printf("%d",sum);
}
```
