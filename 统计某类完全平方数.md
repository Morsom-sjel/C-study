#include <stdio.h>
#include <math.h>

int IsTheNumber ( const int N );

int main()
{
    int n1, n2, i, cnt;
	
    scanf("%d %d", &n1, &n2);
    cnt = 0;
    for ( i=n1; i<=n2; i++ ) {
        if ( IsTheNumber(i) )
            cnt++;
    }
    printf("cnt = %d\n", cnt);

    return 0;
}
int IsTheNumber ( const int N )
{
//统计某类数的完全平方数，该数可能是3位、4位、5位、6位······；
int n,t,i,temp,a[10]={0},flag=0;
n=N;
t=(int)sqrt(n);                //因数学函数的返回值类型是double类型的；而t是int型的；
if(t*t == n)
{
while(n!=0)
{
temp=n%10;
a[temp]++;
n=n/10;
}
for(i=0;i<10;i++)
{
if(a[i]>=2)
{
flag=1;
return 1;
}
}
if(flag==0) return 0;
}
else return 0;
}
